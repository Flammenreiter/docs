=== RLS Beispiel


[source,sql]
----
-- supabase/rls/characters.sql
create policy "Characters are visible to owner and their campaign GM"
on characters for select
using (
  auth.uid() = user_id OR
  auth.uid() = (
    select created_by from campaigns where campaigns.id = characters.campaign_id
  )
);

create policy "Characters can be updated by owner only"
on characters for update
using (auth.uid() = user_id)
with check (auth.uid() = user_id);
----

=== State Management mit Zustand


[source,typescript]
----
// store/characterStore.ts
import { create } from 'zustand';

interface CharacterStore {
  characterId: UUID | null;
  character: Character | null;
  loading: boolean;

  setCharacter: (id: UUID) => Promise<void>;
  updateTrait: (traitId: UUID, level: number) => Promise<void>;
}

export const useCharacterStore = create<CharacterStore>((set) => ({
  characterId: null,
  character: null,
  loading: false,

  setCharacter: async (id: UUID) => {
    set({ loading: true });
    const res = await fetch(`/api/characters/${id}`);
    const character = await res.json();
    set({ character, characterId: id, loading: false });
  },

  updateTrait: async (traitId: UUID, level: number) => {
    const res = await fetch('/api/traits', {
      method: 'POST',
      body: JSON.stringify({ characterId: get().characterId, traitId, level })
    });
    // Refetch character
    await get().setCharacter(get().characterId!);
  }
}));
----


=== API Service


[source,typescript]
----
// services/api.ts
import { useAuth } from '@supabase/auth-helpers-react';

export async function fetchCharacter(id: UUID) {
  const { user } = useAuth();

  const res = await fetch(`/api/characters/${id}`, {
    headers: {
      'Authorization': `Bearer ${user?.session?.access_token}`
    }
  });

  if (!res.ok) throw new Error('Failed to fetch');
  return await res.json();
}

=== Slash Commands Beispiel


[source,typescript]
----
// commands/character.ts
import { SlashCommandBuilder, EmbedBuilder } from 'discord.js';

export const data = new SlashCommandBuilder()
  .setName('character')
  .setDescription('View your character')
  .addStringOption(option =>
    option.setName('name').setDescription('Character name')
  );

export async function execute(interaction) {
  const characterName = interaction.options.getString('name');
  const userId = interaction.user.id;

  // Query Supabase
  const { data: character } = await supabase
    .from('characters')
    .select('*')
    .eq('name', characterName)
    .eq('user_id', userId)
    .single();

  const embed = new EmbedBuilder()
    .setTitle(character.name)
    .addFields(
      { name: 'Level', value: character.level.toString() },
      { name: 'HP', value: `${character.hp_current}/${character.hp_max}` }
    );

  await interaction.reply({ embeds: [embed] });
}
